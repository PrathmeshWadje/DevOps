docker run				-	Create and run a container from an image
docker ps				-	To list currently running containers
docker ps -a				-	-a for all
docker stop <container_name>		-	Start or stop an existing container
docker rm <container_name>		-	Remove a container
docker images				-	List local images
docker rmi <image_name>			-	Delete an Image
docker pull <image_name>		-	Pull an image from a Docker Hub
docker exec				-	Runs a new command in a running container
docker run -d <image_name>		-	Run a container in the background
docker run <container_name>:<version>	-	Install any specific version container
docker run -i				-	Run docker container in interactive mode
docker run -t				-	Allocate a virtual terminal session within the container
docker run -it				-	Run in both interactive and terminal session
docker inspect <container_name>		-	Inspect Environment Variable
